page.includeJSFooter {
	recaptcha {
		disableCompression = 1
		external = 1
		excludeFromConcatenation = 1
	}
	recaptcha = https://www.google.com/recaptcha/api.js?onload=onloadCallbackRecaptcha&render=explicit
}

page.jsFooterInline {
	831 = TEXT
    831.value (
	var onloadCallbackRecaptcha = function() {
		if ($('#g-recaptcha').length > 0) {
			grecaptcha.render('g-recaptcha', {
			  'sitekey' : '{$plugin.tx_pxaformenhancement.settings.siteKey}'
			});
		}
    };
)
}

plugin.tx_pxaformenhancement.settings.siteSecret = {$plugin.tx_pxaformenhancement.settings.siteSecret}
